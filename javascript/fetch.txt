//---------------------------------------------------------------------------------------------- GET

const url = "https://jsonplaceholder.typicode.com/users/1"
fetch(url, {
    method: 'GET',
    headers: {
        'Content-Type': 'application/json'
    },
}).then(res => res.json())
.then(data => {
    console.log(data)
}).catch(error => {
    console.log(error)
})


//---------------------------------------------------------------------------------------------- POST

const url = "https://jsonplaceholder.typicode.com/users"
let data = {
    name: 'aaa',
    lname: 'bbb'
};

fetch(url, {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json'
    },
    body: JSON.stringify(data)
}).then(res => res.json())
.then(data => {
    console.log(data)
}).catch(error => {
    console.log(error)
})
